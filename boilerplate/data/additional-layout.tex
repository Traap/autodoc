% ------------------------------------------------------------------------------
% Used in \useLogo command to include the logo and set the logo to the
% correct size.
% ------------------------------------------------------------------------------
\newcommand{\logoSize}{\includegraphics*[scale=0.60]}%

% local directory
\newcommand{\lLogo}{data/logo.png}%

% boilerplateSpecificDir
\newcommand{\sLogo}{\autodocDir/boilerplateSpecificTestDir/data/logo.png}%

% boilerplateDir
\newcommand{\bLogo}{\autodocDir/boilerplate/data/logo.png}%

\newcommand{\useLogo}{%
  \IfFileExists{\lLogo}{\logoSize{\lLogo}}{%
    \IfFileExists{\sLogo}{\logoSize{\sLogo}}{%
      \IfFileExists{\bLogo}{\logoSize{\bLogo}}{}%
    }%
  }%
}%
