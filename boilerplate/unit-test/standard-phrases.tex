% Standard Unit Test Abstract.

\newcommand{\stdUnitTypeTest}{Unit Test}
\newcommand{\stdUnitTest}[1]{#1 \stdUnitTypeTest}

\newcommand{\stdUnitTestAbstract}[1]{%
  In this document #1 is the \stdCfgItem\ being unit tested and \stdRptPkg\ is
  the #1 \stdRptPkg. #1 is unit tested after this \stdRptPkg\ has been executed,
  test evidence has been obtained, and test evidence supports the conclusion
  \stdSRS\ are satisfied. This document is stored in \stdCompanyName's (Company)
  \stdQMS\ after #1 has been unit tested.
}

% ------------------------------------------------------------------------------
% Standard Unit Test document generator command. 
% ------------------------------------------------------------------------------
\newcommand{\stdUnitTestDoc}[1]{%
  \stdAutoDocLayout{\stdUnitTest{#1}}%
                   {\stdUnitTestAbstract{#1}}%
                   {\unitTestDir/main.tex}%
}%
