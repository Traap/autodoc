% Define column names use by csvreader for the design history file..
\csvnames{dhfNames}%
{1=\proj,2=\pv,3=\type,4=\docnbr,5=\ver,6=\title,7=\pkg,8=\dra,9=\drb,10=\drc,11=\drd,12=\dre}%

% Define a table style for the design history file csvreader.
\csvstyle{dhfStyle}{%
  longtable=|C{0.5cm}|L{4.5cm}|L{9cm}|
  ,table head=\hline \rowcolor{cyan}\# & Document & Title \\ \hline \hline \endhead
  ,late after line=\\\hline
  ,dhfNames
}

% Define a command to select rows from the design history file.
% Parameters:
% #1 - relative path to design history file.
% #2 - project column value 
% #3 - project version column value 
% #4 - document type column valuek
\newcommand{\dhfList}[4]{%
  \IfFileExists{#1}{%
    \begin{center}%
      \csvreader[dhfStyle, separator=pipe, respect all,%
        filter={\equal{\proj}{#2} \and \equal{\pv}{#3} \and \equal{\type}{#4}}%
        ]%
        {#1}{}{\thecsvrow&\docnbr~[\ver]&\title}%
    \end{center}%
  }%
  {\todo[inline]{#1 was not found.}%
  }%
}%
