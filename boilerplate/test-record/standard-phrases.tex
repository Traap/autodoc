\newcommand{\stdTestRecord}{Test Record}

\newcommand{\stdTestRecordAbstract}[1]{%
  In this document #1 is the \stdCfgItem\ being verified and \stdRptPkg\ is the
  #1 \stdRptPkg. #1 is verified after this \stdRptPkg\ has been executed, test
  evidence has been obtained, and test evidence supports the conclusion
  \stdSyRS\ are satisfied. This document is stored in \stdCompanyName's
  (Company) \stdQMS\ after #1 has been verified.
}%
